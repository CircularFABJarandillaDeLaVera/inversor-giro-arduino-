// ════════════════════════════════════════ // INVERSOR DE GIRO - 2 RELÉS + ARDUINO // Módulo relé activo bajo (LOW = activado) // ════════════════════════════════════════ #define RELE1 7 // K1 → giro adelante #define RELE2 8 // K2 → giro atrás #define BTN_FWD 2 // Botón adelante (INPUT_PULLUP) #define BTN_BWD 3 // Botón atrás #define BTN_STP 4 // Botón stop void setup() { pinMode(RELE1, OUTPUT); pinMode(RELE2, OUTPUT); pinMode(BTN_FWD, INPUT_PULLUP); pinMode(BTN_BWD, INPUT_PULLUP); pinMode(BTN_STP, INPUT_PULLUP); // Estado seguro al arrancar motorStop(); Serial.begin(9600); } // ─── Funciones de control ─── void motorFwd() { digitalWrite(RELE2, HIGH); // Apagar K2 primero delay(10); // Tiempo muerto anti-cortocircuito digitalWrite(RELE1, LOW); // Activar K1 (activo bajo) Serial.println("▶ ADELANTE"); } void motorBwd() { digitalWrite(RELE1, HIGH); // Apagar K1 primero delay(10); digitalWrite(RELE2, LOW); // Activar K2 Serial.println("◀ ATRÁS"); } void motorStop() { digitalWrite(RELE1, HIGH); digitalWrite(RELE2, HIGH); Serial.println("■ STOP"); } // ─── Loop principal ─── void loop() { if (digitalRead(BTN_FWD) == LOW) { motorFwd(); delay(250); // Anti-rebote } if (digitalRead(BTN_BWD) == LOW) { motorBwd(); delay(250); } if (digitalRead(BTN_STP) == LOW) { motorStop(); delay(250); } // Control por Serial Monitor (opcional) if (Serial.available()) { char cmd = Serial.read(); if (cmd == 'f') motorFwd(); if (cmd == 'b') motorBwd(); if (cmd == 's') motorStop(); } }

⚠ CONSIDERACIONES CLAVE
  • ⛔ NUNCA actives K1 y K2 a la vez — cortocircuito en la fuente
  • ✅ El código siempre apaga un relé antes de activar el otro (delay 10ms)
  • ✅ GND común obligatorio entre Arduino y fuente del motor
  • ℹ Módulo relé activo bajo: LOW activa, HIGH desactiva
  • ℹ Si usas relés sueltos: añade diodo 1N4007 en antiparalelo en cada bobina
  • ⚡ Los relés deben aguantar la corriente del motor (5A, 10A, según modelo)
  • ✅ Los botones usan INPUT_PULLUP interno del Arduino — no necesitan resistencias externas
  • ℹ Puedes controlar por Serial Monitor: 'f' = adelante, 'b' = atrás, 's' = stop

📦 COMPONENTES NECESARIOS
  • Arduino Uno / Nano / Mega
  • Módulo relé de 2 canales (5V)
  • Motor DC (cualquier voltaje 6–24V)
  • Fuente DC para el motor
  • 3× pulsadores normalmente abiertos
  • Cable dupont macho-hembra
